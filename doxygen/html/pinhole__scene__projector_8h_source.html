314 template < typename _TPoint3D, typename _TPoint2D > void PinholeSceneProjector::
315  projectPoints( const std::vector< _TPoint3D > &scene_pts, std::vector< _TPoint2D > &img_pts,
316  bool check_occlusions, cv::Mat occ_mask ) const
317 {
318  int pts_size = scene_pts.size();
319  img_pts.resize(pts_size);
320 
321  if( check_occlusions )
322  {
323  std::vector<double> scene_pts_z;
324  scene_pts_z.resize(pts_size);
325 
326  if( cam_model_.hasDistCoeff() )
327  {
328  #pragma omp parallel for if( parallelism_enabled_ )
329  for( int i = 0; i < pts_size; i++)
330  {
331  const double scene_pt[3] = { scene_pts[i].x, scene_pts[i].y, scene_pts[i].z };
332  double proj_pt[2], depth;
333  cam_model_.quatRTProject( r_quat_, t_vec_, scene_pt, proj_pt, depth );
334  img_pts[i] = _TPoint2D( proj_pt[0], proj_pt[1]);
335  scene_pts_z[i] = depth;
336  }
337  }
338  else
339  {
340  #pragma omp parallel for if( parallelism_enabled_ )
341  for( int i = 0; i < pts_size; i++)
342  {
343  const double scene_pt[3] = { scene_pts[i].x, scene_pts[i].y, scene_pts[i].z };
344  double proj_pt[2], depth;
345  cam_model_.quatRTProjectWithoutDistortion(r_quat_, t_vec_, scene_pt, proj_pt, depth );
346  img_pts[i] = _TPoint2D( proj_pt[0], proj_pt[1]);
347  scene_pts_z[i] = depth;
348  }
349  }
350 
351  cv::Mat index_mask = cv::Mat_<int>(cam_model_.imgHeight(), cam_model_.imgWidth(), -1);
352  bool has_mask = !occ_mask.empty();
353 
354  for( int i = 0; i < pts_size; i++)
355  {
356  if( img_pts[i].x >= 0 && img_pts[i].y >= 0 &&
357  img_pts[i].x < cam_model_.imgWidth() && img_pts[i].y < cam_model_.imgHeight())
358  {
359  int x = round(img_pts[i].x), y = round(img_pts[i].y);
360  int selected_idx = index_mask.at<int>(y,x);
361  if( selected_idx < 0 )
362  index_mask.at<int>(y,x) = i;
363  else if( scene_pts_z[i] != scene_pts_z[selected_idx] )
364  {
365  // Occlusion
366  if( scene_pts_z[i] < scene_pts_z[selected_idx] )
367  {
368  img_pts[selected_idx] = _TPoint2D( -1, -1 );
369  index_mask.at<int>(y,x) = i;
370  }
371  else
372  img_pts[i] = _TPoint2D( -1, -1 );
373 
374  if( has_mask )
375  occ_mask.at<uchar>(y,x) = 255;
376  }
377  }
378  }
379  }
380  else
381  {
382  if( cam_model_.hasDistCoeff() )
383  {
384  #pragma omp parallel for if( parallelism_enabled_ )
385  for( int i = 0; i < pts_size; i++)
386  {
387  const double scene_pt[3] = { scene_pts[i].x, scene_pts[i].y, scene_pts[i].z };
388  double proj_pt[2];
389  cam_model_.quatRTProject( r_quat_, t_vec_, scene_pt, proj_pt );
390  img_pts[i] = _TPoint2D( proj_pt[0], proj_pt[1]);
391  }
392  }
393  else
394  {
395  #pragma omp parallel for if( parallelism_enabled_ )
396  for( int i = 0; i < pts_size; i++)
397  {
398  const double scene_pt[3] = { scene_pts[i].x, scene_pts[i].y, scene_pts[i].z };
399  double proj_pt[2];
400  cam_model_.quatRTProjectWithoutDistortion(r_quat_, t_vec_, scene_pt, proj_pt );
401  img_pts[i] = _TPoint2D( proj_pt[0], proj_pt[1]);
402  }
