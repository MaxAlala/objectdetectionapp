218 void PatchMatchFlow :: createFilter()
219 {
220  std::vector<double> filt;
221  filt.resize(aperture_);
222 
223  g_filter_.resize( aperture_*aperture_ );
224  double s = 0.;
225  for (int x = -h_aperture_, i = 0; x <= h_aperture_; x++, i++)
226  {
227  filt[i] = (float)exp(-(double(x)*double(x))/(2*g_filter_sigma_*g_filter_sigma_));
228  s += filt[i];
229  }
230 
231  s = 1./s;
232  for (int i = 0; i < aperture_; i++)
233  {
234  filt[i] *= s;
235  }
236 
237  s = 0;
238  for (int i = 0; i < aperture_; i++)
239  {
240  for (int j = 0; j < aperture_; j++)
241  {
242  g_filter_[aperture_*j + i] = filt[i]*filt[j];
243  s += g_filter_[aperture_*j + i];
244  }
245  }
246 
247  s = 1./s;
248  for (int i = 0; i < aperture_*aperture_; i++)
249  {
250  g_filter_[i] *= s;

272 template< typename _T > inline
273  double PatchMatchFlow::computeCost( const cv::Mat &bimg0, const cv::Mat &bimg1,
274  cv::Point &p0, cv::Point &p1 )
275 {
276  double cost = 0;
277  for( int y = 0; y < aperture_; y++)
278  {
279  const _T *img0_p = bimg0.ptr<_T>(p0.y - h_aperture_),
280  *img1_p = bimg1.ptr<_T>(p1.y - h_aperture_);
281 
282  img0_p += (p0.x - h_aperture_);
283  img1_p += (p1.x - h_aperture_);
284  int filt_idx = aperture_*y;
285  for( int x = 0; x < aperture_; x++, img0_p++, img1_p++, filt_idx++ )
286  cost += g_filter_[filt_idx]*pixDist(*img0_p, *img1_p);
287  }
288 
289  return cost;
290 }
