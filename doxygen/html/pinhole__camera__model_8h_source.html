715 template < typename _T >
716  inline void PinholeCameraModel::denormalize( const _T src_pt[2],
717  _T dest_pt[2] ) const
718 {
719  const _T &x = src_pt[0], &y = src_pt[1];
720 
721  _T x2 = x * x, y2 = y * y, xy_dist, r2, r4, r6;
722  _T l_radial, h_radial, radial, tangential_x, tangential_y;
723 
724  r2 = x2 + y2;
725  r6 = r2 * ( r4 = r2 * r2 );
726 
727  l_radial = _T(1.0) + _T(dist_k0_) * r2 + _T(dist_k1_) * r4 + _T(dist_k2_) * r6;
728  h_radial = _T(1.0)/(_T(1.0) + _T(dist_k3_)*r2 + _T(dist_k4_)*r4 + _T(dist_k5_)*r6);
729  radial = l_radial * h_radial;
730  xy_dist = _T(2.0) * x * y;
731  tangential_x = xy_dist * _T(dist_px_) + _T(dist_py_) * ( r2 + _T(2.0) * x2 );
732  tangential_y = xy_dist * _T(dist_py_) + _T(dist_px_) * ( r2 + _T(2.0) * y2 );
733 
734  _T x_dist = x * radial + tangential_x;
735  _T y_dist = y * radial + tangential_y;
736 
737  dest_pt[0] = x_dist * _T(fx_) + _T(cx_);
738  dest_pt[1] = y_dist * _T(fy_) + _T(cy_);
739 }

749 template < typename _T >
750  inline void PinholeCameraModel::normalize( const _T src_pt[2],
751  _T dest_pt[2] ) const
752 {
753  _T dist_pt[2], prev_pt[2], f[2], j[4];
754 
755  dist_pt[0] = ( src_pt[0] - _T(cx_) ) * _T(inv_fx_);
756  dist_pt[1] = ( src_pt[1] - _T(cy_) ) * _T(inv_fy_);
757 
758  prev_pt[0] = dest_pt[0] = dist_pt[0];
759  prev_pt[1] = dest_pt[1] = dist_pt[1];
760 
761  for(int iter = 0; iter < 10; iter++)
762  {
763  computeNormalizationFunction( dist_pt, dest_pt, f, j );
764 
765  _T tmp_u = f[0] - j[0]*dest_pt[0] - j[1]*dest_pt[1];
766  _T tmp_v = f[1] - j[2]*dest_pt[0] - j[3]*dest_pt[1];
767 
768  j[0] = _T(1.0) - j[0];
769  j[1] = -j[1];
770  j[2] = -j[2];
771  j[3] = _T(1.0) - j[3];
772 
773  invertMat2X2( j );
774 
775  dest_pt[0] = j[0]*tmp_u + j[1]*tmp_v;
776  dest_pt[1] = j[2]*tmp_u + j[3]*tmp_v;
777 
778  // Exit if convergence
779  if( std::abs<_T>(prev_pt[0] - dest_pt[0]) < _T(term_epsilon_) &&
780  std::abs<_T>(prev_pt[1] - dest_pt[1]) < _T(term_epsilon_) )
781  break;
782 
783  prev_pt[0] = dest_pt[0];
784  prev_pt[1] = dest_pt[1];
785  }
786 }

1007 template < typename _T > inline void PinholeCameraModel::invertMat2X2( _T m[4] ) const
1008 {
1009  _T inv_det = 1.0/(m[0]*m[3] - m[1]*m[2]), tmp = m[0];
1010  m[0] = inv_det*m[3];
1011  m[1] *= -inv_det;

1038 template < typename _T >
1039  inline void PinholeCameraModel::computeNormalizationFunction( const _T dist_pt[2], _T cur_pt[2],
1040  _T f[2], _T j[4] ) const
1041 {
1042  _T &x = cur_pt[0], &y = cur_pt[1];
1043  const _T &dist_x = dist_pt[0], &dist_y = dist_pt[1];
1044 
1045  _T x2 = x * x, y2 = y * y,
1046  xyd, r2, r4, r6, l_radial,// h_radial,
1047  il_radial, il_radial2, ih_radial, tangential_x, tangential_y,
1048  d_l_radial_dx, d_l_radial_dy, d_ih_radial_dx, d_ih_radial_dy,
1049  d_tangential_x_dx, d_tangential_x_dy, d_tangential_y_dx, d_tangential_y_dy;
1050 
1051  r2 = x2 + y2;
1052  r6 = r2 * ( r4 = r2 * r2 );
1053 
1054  l_radial = _T(1.0) + _T(dist_k0_)*r2 + _T(dist_k1_) * r4 + _T(dist_k2_) * r6;
1055  ih_radial = _T(1.0) + _T(dist_k3_)*r2 + _T(dist_k4_)*r4 + _T(dist_k5_)*r6;
1056 
1057  il_radial = _T(1.0)/l_radial;
1058 // h_radial = _T(1.0)/ih_radial;
1059 
1060  il_radial2 = _T(1.0)/(l_radial*l_radial);
1061 
1062  d_l_radial_dx = _T(6.0)*_T(dist_k2_)*x*r4 + _T(4.0)*_T(dist_k1_)*x*r2 + _T(2.0)*_T(dist_k0_)*x;
1063  d_l_radial_dy = _T(6.0)*_T(dist_k2_)*y*r4 + _T(4.0)*_T(dist_k1_)*y*r2 + _T(2.0)*_T(dist_k0_)*y;
1064 
1065  d_ih_radial_dx = _T(6.0)*_T(dist_k5_)*x*r4 + _T(4.0)*_T(dist_k4_)*x*r2 + _T(2.0)*_T(dist_k3_)*x;
1066  d_ih_radial_dy = _T(6.0)*_T(dist_k5_)*y*r4 + _T(4.0)*_T(dist_k4_)*y*r2 + _T(2.0)*_T(dist_k3_)*y;
1067 
1068  xyd = _T(2.0)*x*y;
1069  tangential_x = xyd*_T(dist_px_) + _T(dist_py_)*( r2 + _T(2.0)*x2 );
1070  tangential_y = xyd*_T(dist_py_) + _T(dist_px_)*( r2 + _T(2.0)*y2 );
1071 
1072  d_tangential_x_dx = _T(2.0)*_T(dist_px_)*y + _T(6.0)*_T(dist_py_)*x;
1073  d_tangential_x_dy = d_tangential_y_dx = _T(2.0)*_T(dist_py_)*y + _T(2.0)*_T(dist_px_)*x;
1074  d_tangential_y_dy = _T(6.0)*_T(dist_px_)*y + _T(2.0)*_T(dist_py_)*x;
1075 
1076  f[0] = ( dist_x - tangential_x )*ih_radial*il_radial;
1077  f[1] = ( dist_y - tangential_y )*ih_radial*il_radial;
1078 
1079  j[0] = d_ih_radial_dx*( dist_x - tangential_x )*il_radial +
1080  -d_tangential_x_dx*ih_radial*il_radial -
1081  ( dist_x - tangential_x )*d_l_radial_dx*ih_radial*il_radial2;
1082 
1083  j[1] = d_ih_radial_dy*( dist_x - tangential_x )*il_radial +
1084  -d_tangential_x_dy* ih_radial*il_radial -
1085  d_l_radial_dy*( dist_x - tangential_x )*ih_radial*il_radial2;
1086 
1087  j[2] = d_ih_radial_dx*( dist_y - tangential_y )*il_radial +
1088  -d_tangential_y_dx * ih_radial * il_radial -
1089  d_l_radial_dx*( dist_y - tangential_y )*ih_radial*il_radial2;
1090 
1091  j[3] = d_ih_radial_dy*(dist_y - tangential_y )*il_radial +
1092  -d_tangential_y_dy * ih_radial * il_radial -
1093  d_l_radial_dy*(dist_y- tangential_y )*ih_radial*il_radial2;
